\subsection*{Definitely assigns}

\begin{figure}
\begin{mathpar}
\inferrule*[right=assign]
{
  \wf{\Gamma}{e}
}
{
  \wfStmt{\Gamma}{x = e}{\{x\}}
}
\and
\inferrule*[right=seq]
{
  \wfStmt{\Gamma}{s_1}{\Delta_1} \\
  \wfStmt{\Gamma \cup \Delta_1}{s_2}{\Delta_2}
}
{
  \wfStmt{\Gamma}{s_1\;s_2}{\Delta_1 \cup \Delta_2}
}
\and
\inferrule*[right=if]
{
  \wf{\Gamma}{e} \\
  \wfStmt{\Gamma}{s}{\Delta_0} \\
  \wf{\Gamma}{e_i} \\
  \wfStmt{\Gamma}{s_i}{\Delta_i} \quad (\text{for } i \in \{1, \ldots, n\})
  \\
}
{
  \wfStmt{\Gamma}{\texttt{if}\;e:\;s\;\texttt{elif}\;e_1:\;s_1\;\cdots\;\texttt{elif}\;e_n:\;s_n}{\varnothing}
}
\and
\inferrule*[right=if\_else]
{
  \wf{\Gamma}{e} \\
  \wfStmt{\Gamma}{s}{\Delta} \\
  \wf{\Gamma_1}{s_1} \\
  \cdots \\
  \wf{\Gamma_n}{s_n} \\
  \wf{\Gamma'}{s'}
}
{
  \wfStmt{\Gamma}
         {\texttt{if}\;e:\;s\;\texttt{elif}\;e_1:\;s_1\;\cdots\;\texttt{elif}\;e_n:\;s_n\;\texttt{else}\;e':\;s'}
}
\end{mathpar}
\caption{Well-formed statements}
\end{figure}
