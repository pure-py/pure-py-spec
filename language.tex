\section*{Judgements}

\subsection*{DefinitelyRet}
\[
\inferrule*[right=return]
{}
{\DefRet{\texttt{return}\;e}}
\]

\[
\inferrule*[right=seq\_1]
{\DefRet{s_1}}
{\DefRet{s_2\;s_2}}
\qquad
\inferrule*[right=seq\_2]
{\DefRet{s_2}}
{\DefRet{s_2\;s_2}}
\]

\[
\inferrule*[right=if\_else]
{\DefRet{s} \\
 \DefRet{s_1} \\
 \cdots \\
 \DefRet{s_n} \\
 \DefRet{s'}}
{\DefRet{\texttt{if}\;e:\;s\;\texttt{elif}\;e_1:\;s_1\;\cdots\;\texttt{elif}\;e_n:\;s_n\;\texttt{else}\;e':\;s'}}
\]

\subsection*{DefinitelyAssigns}
\[
\inferrule*[right=assign]
{}
{\DefAssigns{\varepsilon, x}{x = e}}
\]

\[
\inferrule*[right=if\_else]
{\DefAssigns{\Gamma}{s} \\
 \DefAssigns{\Gamma_1}{s_1} \\
 \cdots \\
 \DefAssigns{\Gamma_n}{s_n} \\
 \DefAssigns{\Gamma'}{s'}}
{\DefAssigns{\mathrm{intersection}\;\Gamma\;\Gamma_1\;\cdots\;\Gamma_n\;\Gamma'}
             {\texttt{if}\;e:\;s\;\texttt{elif}\;e_1:\;s_1\;\cdots\;\texttt{elif}\;e_n:\;s_n\;\texttt{else}\;e':\;s'}}
\]
